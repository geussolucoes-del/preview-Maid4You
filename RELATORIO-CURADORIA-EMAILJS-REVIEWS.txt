// ── PROGRESS ─────────────────────────────────────────────────────
function updateProgress(step) {
  for (let i = 1; i <= 3; i++) {
    const num = document.getElementById('psn'+i);
    const item = document.getElementById('psi'+i);
    num.classList.remove('active','done');
    item.classList.remove('active','done');
    if (i < step) { num.classList.add('done'); item.classList.add('done'); }
    if (i === step) { num.classList.add('active'); item.classList.add('active'); }
    if (i < 3) {
      const line = document.getElementById('psl'+i);
      line.classList.toggle('done', i < step);
    }
  }
}

// ── PILL / YN SELECTORS ──────────────────────────────────────────
function selectPill(field, val, btn) {

  btn.classList.add('selected');
  document.getElementById(field).value = val;

// ── VALIDATION ────────────────────────────────────────────────────
function validateStep1() {
  const fields = [
    { id: 'firstName', check: v => v.trim().length > 1 },
    { id: 'lastName',  check: v => v.trim().length > 1 },
    { id: 'phone',     check: v => v.replace(/\D/g,'').length >= 10 },
    { id: 'email',     check: v => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v) },
    { id: 'zipCode',   check: v => /^\d{5}$/.test(v.replace(/\D/g,'')) },
  ];
  let ok = true;
  fields.forEach(f => {
    const el = document.getElementById(f.id);
    const valid = f.check(el.value);
    el.classList.toggle('error', !valid);
    el.classList.toggle('valid', valid);
    if (!valid) ok = false;
  });
  if (!document.getElementById('contactPref').value) {
    alert('Please select your preferred contact method.');
    ok = false;
  }
  return ok;
}
function validateStep2() {
  const fields = ['serviceType','floors','bedrooms','bathrooms','frequency'];
  let ok = true;
  fields.forEach(id => {
    const el = document.getElementById(id);
    const valid = el.value !== '';
    el.classList.toggle('error', !valid);
    el.classList.toggle('valid', valid);
    if (!valid) ok = false;
  });
  if (!document.getElementById('pets').value) {
    alert('Please indicate if you have pets.');
    ok = false;
  }
  if (!document.getElementById('prefDate').value) {
    alert('Please select a preferred start date.');
    ok = false;
  }
  return ok;
}
function validateStep3() {
  const el = document.getElementById('hearAbout');
  const valid = el.value !== '';
  el.classList.toggle('error', !valid);
  el.classList.toggle('valid', valid);
  return valid;
}

// ── NAVIGATION ────────────────────────────────────────────────────

  currentStep = from + 1;
  document.getElementById('step'+currentStep).classList.add('active');

  document.querySelector('.qright').scrollTo({ top: 0, behavior: 'smooth' });
}
function prevStep(from) {

  document.getElementById('step'+currentStep).classList.add('active');

}

    lastName:    document.getElementById('lastName').value,

    pets:        document.getElementById('pets').value,

    prefDate:    document.getElementById('prefDate').value,
    hearAbout:   document.getElementById('hearAbout').value,
    notes:       document.getElementById('notes').value,
  };

  // Show sending
  document.getElementById('step3').classList.remove('active');
  document.getElementById('sendingState').style.display = 'block';

  const emailPayload = {
    to_email: 'maid44youcleaningservice@gmail.com',

    from_name: `${data.firstName} ${data.lastName}`.trim(),
    first_name: data.firstName,
    last_name: data.lastName,
    phone: data.phone,
    email: data.email,
    zip_code: data.zipCode,
    contact_preference: data.contactPref,
    service_type: data.serviceType,
    floors: data.floors,
    bedrooms: data.bedrooms,
    bathrooms: data.bathrooms,
    pets: data.pets,
    frequency: data.frequency,
    preferred_start_date: data.prefDate,
    hear_about: data.hearAbout,
    notes: data.notes,
    message: `New Free Estimate request from ${data.firstName} ${data.lastName}

Phone / Text: ${data.phone}
Email: ${data.email}
ZIP Code: ${data.zipCode}
Preferred Contact: ${data.contactPref}
Service Type: ${data.serviceType}
Floors: ${data.floors}
Bedrooms: ${data.bedrooms}
Bathrooms: ${data.bathrooms}
Pets: ${data.pets}
Frequency: ${data.frequency}
Preferred Start Date: ${data.prefDate}
Heard About Us: ${data.hearAbout}
Notes: ${data.notes || '—'}`
  };

  const finishSuccess = () => {
    try {
      sessionStorage.setItem('m4y_form_submitted', '1');
      sessionStorage.setItem('m4y_last_submission', JSON.stringify(emailPayload));
    } catch(e) {}
    window.location.href = 'thankyou.html';
  };

  const finishError = (err) => {
    console.error('EmailJS send error:', err);
    document.getElementById('sendingState').style.display = 'none';

  if (!window.M4Y_EMAILJS_READY || !window.emailjs) {
    finishError(new Error('EmailJS not initialized'));
    return;
  }

  emailjs.send(EMAILJS_CONFIG.serviceId, EMAILJS_CONFIG.templateId, emailPayload)
    .then(finishSuccess)
    .catch(finishError);
}

document.getElementById('phone').addEventListener('input', function() {

  if (v.length >= 7) v = '(' + v.slice(0,3) + ') ' + v.slice(3,6) + '-' + v.slice(6);

});

<script>
document.addEventListener("click", function(e){
  const smsLink = e.target.closest('a[href^="sms:"]');
  if(!smsLink) return;

  window.m4yPushEvent && window.m4yPushEvent("click_to_text", {
    click_url: smsLink.getAttribute("href"),
    click_text: (smsLink.textContent || "").trim(),
    page_path: window.location.pathname
  });
});
</script>


<script>
/* === CENTER ESTIMATE FORM ON ENTRY ===
   Quando o usuário vem de Free Estimate, centraliza o formulário suavemente. */
(function(){
  function centerEstimateForm(){
    if (window.location.hash !== '#formContainer') return;
    var target = document.getElementById('formContainer');
    if (!target) return;
    setTimeout(function(){
      target.scrollIntoView({ behavior: 'smooth', block: 'center', inline: 'nearest' });
    }, 220);
  }
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', centerEstimateForm);
  } else {
    centerEstimateForm();
  }
})();
</script>
